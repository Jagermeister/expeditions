class Deck:
    @staticmethod
    def create(cardCount):
        return 2**cardCount - 1

    @staticmethod
    def cardPeek(container, cardCount):
        unsetIndex = random.randrange(cardCount)
        while unsetIndex:
            container &= container - 1
            unsetIndex -= 1
        return len(bin(container)) - len(bin(container).rstrip('0'))
